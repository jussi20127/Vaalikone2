-- Create table for Parties

CREATE TABLE PARTIES (
   	ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
   	PUOLUE VARCHAR(50) NOT NULL)
ENGINE=INNODB;

-- Create table for Candidates

CREATE TABLE CANDIDATES (
	ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	SUKUNIMI VARCHAR(25) NOT NULL,
	ETUNIMI VARCHAR(25) NOT NULL,
	NUMERO INTEGER NOT NULL UNIQUE,
	KOTIPAIKKAKUNTA VARCHAR(25) NOT NULL,
	PUOLUE VARCHAR(25) NOT NULL,
	IKA INTEGER NOT NULL,
	AMMATTI VARCHAR(50) NOT NULL,
	PUOLUE_ID INT NOT NULL,
	FOREIGN KEY(PUOLUE_ID) REFERENCES PARTIES(ID)
)
ENGINE=INNODB;

-- Create table for Questions

CREATE TABLE QUESTIONS (
	ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	KYSYMYS VARCHAR(150) NOT NULL,
	AIHEALUE VARCHAR(25) NOT NULL)
ENGINE=INNODB;

-- Create table for Subjects

CREATE TABLE SUBJECTS (
	ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
	AIHEALUE VARCHAR(25) NOT NULL)
ENGINE=INNODB;



-- Create table for Answers

CREATE TABLE ANSWERS (
	ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	VASTAUS INTEGER,
	CANDIDATE_ID INT,
	SUBJECT_ID INT,
	QUESTION_ID INT,
	FOREIGN KEY(CANDIDATE_ID) REFERENCES CANDIDATES(ID),
	FOREIGN KEY(SUBJECT_ID) REFERENCES SUBJECTS(ID),
	FOREIGN KEY(QUESTION_ID) REFERENCES QUESTIONS(ID)
	
)
ENGINE=INNODB;
